Exemplos de Funções em JavaScript

1️⃣ Função sem retorno (`void`)

Essa função apenas executa uma ação sem retornar nada.


function dizerOla(nome) {
    console.log("Olá, " + nome + "!");
}

dizerOla("Lucas"); // Saída: Olá, Lucas!
Olá, visitante!



2️⃣ Função com retorno

Essa função retorna um valor quando chamada.


            function dobrar(numero) {
                return numero * 2;
            }
            
            let resultado = dobrar(4);
            console.log(resultado); // Saída: 8
Olá, visitante!

3️⃣ Função anônima (atribuída a uma variável)


const somar = function(a, b) {
    return a + b;
};

console.log(somar(5, 7)); // Saída: 12

4️⃣ Arrow Function (Função de seta)

Uma forma mais curta de escrever funções.


const multiplicar = (x, y) => x * y;

console.log(multiplicar(3, 4)); // Saída: 12

5️⃣ Função que recebe outra função como argumento (Callback)


function processarNumero(numero, callback) {
    return callback(numero);
}

function aoQuadrado(n) {
    return n * n;
}

console.log(processarNumero(5, aoQuadrado)); // Saída: 25

6️⃣ Função que retorna outra função (Função de Alta Ordem)


function criarMultiplicador(multiplicador) {
    return function(numero) {
        return numero * multiplicador;
    };
}

const dobrarNumero = criarMultiplicador(2);
console.log(dobrarNumero(5)); // Saída: 10
